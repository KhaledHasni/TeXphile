% ***************** Profile URLs *****************
\edef\MyLinkedinProfileURL{https://\MyLinkedinProfile}
\edef\MyGithubProfileURL{https://\MyGithubProfile}

% Actual resume header starts here
\begin{center}
    % *Full name*
    % Centered, in huge letters (2nd largest predefined font size in LaTeX).
    % All letters are capitalized with initials having a larger size than the rest if SmallCaps is set to "y".
    {\Huge \ifthenelse{\equal{\SmallCaps}{y}}{\scshape}{} \MyFullName} \\

    \vspace{1pt}

    % *Job title / active position*
    % Centered, standard font, fontsize set to 10pt.
    % 2nd arg is the baselineskip i.e the vertical spacing between lines: Not being used here since this entry will only take up a single line, but it's necessary for the fontsize's syntax.
    % Note: LaTeX recommends a baselineskip of fontsize * 1.2
    {\fontsize{10pt}{12pt}\selectfont \textbf{\MyJobTitle}} \\

    \vspace{2pt}

    % *Phone number entry*
    % Standard font, fontsize is set to 10pt.
    % Inserts a phone icon from the FontAwesome package (if UseIcons is set to "y") and vertically aligns it with the text.
    % Entry can be removed if PhoneNbr is set to "n".
    \makebox[0.2\textwidth][c]{
    \ifthenelse{\equal{\PhoneNbr}{y}}{
        {\small
        \ifthenelse{\equal{\UseIcons}{y}}{
            \raisebox{-0.025\height}\faPhone
        }{}
        {\fontsize{10pt}{12pt}\selectfont \MyPhoneNumber}
        }
    \hspace{-5.5mm}
    }{}  

    % *E-mail entry*
    % Standard font, fontsize is set to 10pt.
    % Inserts an email icon from the FontAwesome package (if UseIcons is set to "y") and vertically aligns it with the text.
    % Creates a hyperlink to the provided email address.
    % Entry can be removed if EmailAddr is set to "n".
    \ifthenelse{\equal{\EmailAddr}{y}}{
        \ifthenelse{\equal{\UseIcons}{y}}{
            \href{mailto:\MyEmailAddress}{\raisebox{-0.035\height}\faEnvelope\ {\fontsize{10pt}{12pt}\selectfont \MyEmailAddress}}
        }
        %else
        {\href{mailto:\MyEmailAddress}{\fontsize{10pt}{12pt}\selectfont \MyEmailAddress}}
    \hspace{-5.5mm}
    }{}

    % *Linkedin entry*
    % Standard font, fontsize is set to 10pt.
    % Inserts Linkedin's icon from the FontAwesome package (if UseIcons is set to "y") and vertically aligns it with the text.
    % Creates a hyperlink to the provided Linkedin profile.
    % Entry can be removed if LinkedinProf is set to "n".
    \ifthenelse{\equal{\LinkedinProf}{y}}{
        \ifthenelse{\equal{\UseIcons}{y}}{
            \href{\MyLinkedinProfileURL}{\raisebox{-0.035\height}\faLinkedin\ {\fontsize{10pt}{12pt}\selectfont \MyLinkedinProfile}}
        }
        %else
        {\href{\MyLinkedinProfileURL}{\fontsize{10pt}{12pt}\selectfont \MyLinkedinProfile}}
    \hspace{-5.5mm}
    }{}

    % *Github entry*
    % Standard font, fontsize is set to 10pt.
    % Inserts Github's icon from the FontAwesome package (if UseIcons is set to "y") and vertically aligns it with the text.
    % Creates a hyperlink to the provided Github profile.
    % Entry can be removed if GithubProf is set to "n".
    \ifthenelse{\equal{\GithubProf}{y}}{
        \ifthenelse{\equal{\UseIcons}{y}}{
            \href{\MyGithubProfileURL}{\raisebox{-0.035\height}\faGithub\ {\fontsize{10pt}{12pt}\selectfont \MyGithubProfile}}
        }
        %else
        {\href{\MyGithubProfileURL}{\fontsize{10pt}{12pt}\selectfont \MyGithubProfile}}
    }{}
    }

    % Add an 8 point vertical spacing at the end of the header.
    \vspace{-8pt}
\end{center}
